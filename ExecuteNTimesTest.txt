$IN
DECLARE index;
DECLARE sum;
sum = 0;
index = 0;
EXECUTE 10 TIMES {
	sum = sum + index;
	index = index + 1;
};
PRINT index;
PRINT sum;
$OUT
45
9

$IN
DECLARE index;
DECLARE innerIndex;
DECLARE sum;
DECLARE x;
sum = 0;
index = 0;
x = 10;
EXECUTE x TIMES {
    innerIndex = 0;
	EXECUTE index TIMES {
		sum = sum + innerIndex;
		innerIndex = innerIndex + 1;
	}
	index = index + 1;
}
PRINT index;
PRINT innerIndex;
PRINT sum;
$OUT
10
9
120
